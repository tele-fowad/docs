#### Request Body Parameters

<ParamField path="wabaid" type="string" required>
  Specifies the whatsapp business account id.
</ParamField>

```json SampleRequest.json
{

  "language": "en_US",
  "category": "authentication",
  "message_send_ttl_seconds": 60,
  "components": [
    {
      "type": "body",
      "add_security_recommendation": true
    },
    {
      "type": "footer",
      "code_expiration_minutes": 5
    },
    {
      "type": "buttons",
      "buttons": [
        {
          "type": "otp",
          "otp_type": "copy_code",
          "text": "Copy Code"
        }
      ]
    }
  ]
}
```

```json SampleResponse.json
{
  "id": "123412341234123",
  "status": "APPROVED",
  "category": "AUTHENTICATION"
}
```
